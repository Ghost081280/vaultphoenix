Quick Start Example

App.js 
import { VaultPhoenixSDK } from '@vaultphoenix/sdk';

// Initialize SDK
const phoenix = new VaultPhoenixSDK({
  apiKey: 'your_api_key_here',
  environment: 'production'
});

// Start AR session
const startARSession = async () => {
  try {
    await phoenix.startARSession({
      onCoinFound: (coin) => {
        console.log('Found coin:', coin);
      },
      onCollected: (reward) => {
        console.log('Collected reward:', reward);
      }
    });
  } catch (error) {
    console.error('AR session error:', error);
  }
};
ViewController.swift 
import VaultPhoenixSDK

// Initialize SDK
let phoenix = VaultPhoenixSDK(
    apiKey: "your_api_key_here",
    environment: .production
)

// Start AR session
phoenix.startARSession { result in
    switch result {
    case .success(let session):
        session.onCoinFound = { coin in
            print("Found coin: \(coin)")
        }
        session.onCollected = { reward in
            print("Collected reward: \(reward)")
        }
    case .failure(let error):
        print("AR session error: \(error)")
    }
}
MainActivity.kt
